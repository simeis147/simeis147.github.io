    class Solution {
    public int evalRPN(String[] tokens) {
        Deque<Integer> stack = new LinkedList();
        for (String s : tokens) {
            if ("+".equals(s)) {        // leetcode ÂÜÖÁΩÆjdkÁöÑÈóÆÈ¢òÔºå‰∏çËÉΩ‰ΩøÁî®==Âà§Êñ≠Â≠óÁ¨¶‰∏≤ÊòØÂê¶Áõ∏Á≠â
                stack.push(stack.pop() + stack.pop());      // Ê≥®ÊÑè - Âíå/ ÈúÄË¶ÅÁâπÊÆäÂ§ÑÁêÜ
            } else if ("-".equals(s)) {
                stack.push(-stack.pop() + stack.pop());
            } else if ("*".equals(s)) {
                stack.push(stack.pop() * stack.pop());
            } else if ("/".equals(s)) {
                int temp1 = stack.pop();
                int temp2 = stack.pop();
                stack.push(temp2 / temp1);
            } else {
                stack.push(Integer.valueOf(s));
            }
        }
        return stack.pop();
    }
}

    public int[] maxSlidingWindow(int[] nums, int k) {
