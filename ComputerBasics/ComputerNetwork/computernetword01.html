                      public static boolean isIpv4Address(String input) {
    if (input == null || input.isEmpty()) {
        return false;
    }
    String[] parts = input.split("\\.");
    if (parts.length != 4) {
        return false;
    }
    for (String part : parts) {

            return false;
        }
    }
    return true;
}
